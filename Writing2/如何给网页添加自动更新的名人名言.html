'天才就其本质而论只不过是对事业、对工作过程的热爱而已。——高尔基',
'勤能补拙是良训，一分辛劳一分才。——华罗庚',
'知识就是力量。——培根',
'知识是一种快乐，而好奇则是知识的萌芽。——培根',
]

var randomNumber1 = Math.floor(Math.random()*(quotes.length));//用Math.random产生0-1的随机数，乘以quotes的个数以获取quotes的随机序号；用Math.floor来取整后返回给变量randomNumber1
document.getElementById('MyQuotes').innerHTML = quotes[randomNumber1];//用getElementById通过id找到赋值目标，用innerHTML来将quotes中的序号为"randomNumber1"的量进行赋值
