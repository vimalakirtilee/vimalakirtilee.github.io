如何给网页添加自动更新的名人名言

<!DOCTYPE html>
<html>
<head>
<title>Quote Gen</title> <!--网页题目-->
</head>

<body>
<!--本例设置了两个产生名人名言的方式，第一种方式为自动产生名人名言；第二种为点击按钮后产生新的名人名言-->
<!--以下为第一种方式-->
<h1>My Quotes</h1> <!--此处为一级标题-->
<div id="MyQuotes"></div> <!--<div>是一种分段方式-->

<!--以下为第二种方式-->
<h1>Simple Quote Generator</h1> 
<div id="quoteDisplay">
<!--QUotes will display here-->
</div>
<button onclick="newQuote()">New Quote</button> <!--此处设置了一个按钮，通过按钮来调用"newQuote"函数，产生新的名言-->


<!--以下为js的部分-->
<script scr="javascript.js">
<!--创建一个变量"quotes"记录名言-->
var quotes = [
'天才就其本质而论只不过是对事业、对工作过程的热爱而已。——高尔基',
'勤能补拙是良训，一分辛劳一分才。——华罗庚',
'知识就是力量。——培根',
'知识是一种快乐，而好奇则是知识的萌芽。——培根',
]

var randomNumber1 = Math.floor(Math.random()*(quotes.length));//用Math.random产生0-1的随机数，乘以quotes的个数以获取quotes的随机序号；用Math.floor来取整后返回给变量randomNumber1
document.getElementById('MyQuotes').innerHTML = quotes[randomNumber1];//用getElementById通过id找到赋值目标，用innerHTML来将quotes中的序号为"randomNumber1"的量进行赋值

<!--将上述命令放入newQuote函数中，-->
function newQuote() {
var randomNumber2 = Math.floor(Math.random()*(quotes.length));
document.getElementById('quoteDisplay').innerHTML = quotes[randomNumber2];
}

</script>
</body>
</html>
